

//将文件上传到已经关联的gitHub中
1：git clone https://github.com/zhou-X-boy/Git_study.git 
2：cd Git_study
2：git add 文件名  
3：git commit -m "备注"
4：git push origin master



//将本地文件夹上传到还未关联的gitHub中
1：git init
2：git add .
3：git commit -m "备注"
4：git remote add origin https://github.com/zhou-X-boy/Node_study.git
5：git push -u origin master



//更新项目后上传更新了的内容到gitHub中
1：git commit          		 //显示修改过的文件
2：git commit -a  			//提交所有的修改
按i可以进行修改，添加描述文件
修改完毕后按 alt+c 退出输入模式
输入 :wq 退出修改模式
3：git commit -a -m "这是描述"
4：git push 			//提交到服务器
